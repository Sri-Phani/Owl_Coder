	#define mod 1000000007 
	int TotalWays(int N)
	{
	    // Code here
	    //if(N==1) return 4;
	    long long a=1,b=1,c=a+b;
	    for(int i=2;i<=N;i++)
	    {
	        a=b;//1
	        b=c;//2
	        c=(a+b)%mod;
	    }
	    return (c*c)%mod;
	    
	    
	}
